-- Create City table
CREATE TABLE City (
    City_ID INTEGER PRIMARY KEY,
    Name TEXT NOT NULL
);

-- Create Barangay table
CREATE TABLE Barangay (
    Barangay_ID INTEGER PRIMARY KEY,
    Name TEXT NOT NULL,
    City_ID INTEGER,
    Color TEXT,
    FOREIGN KEY (City_ID) REFERENCES City(City_ID)
);

--Create Item Table
CREATE TABLE Item (
    Item_ID INTEGER PRIMARY KEY,
    Image TEXT,
    Title TEXT NOT NULL,
    Item_Type TEXT,
    POS_Item INTGER,
    Reorder INTEGER,
    Deliver INTEGER,
    Pickup INTEGER,
    Buy INTEGER,
    Date DATE
);
