-- Create City table
CREATE TABLE City (
    City_ID INTEGER PRIMARY KEY,
    Name TEXT NOT NULL
);

-- Create Barangay table
CREATE TABLE Barangay (
    Barangay_ID INTEGER PRIMARY KEY,
    Name TEXT NOT NULL,
    City_ID INTEGER,
    Color TEXT,
    FOREIGN KEY (City_ID) REFERENCES City(City_ID)
);

--Create Item Table
CREATE TABLE Item (
    Item_ID INTEGER PRIMARY KEY,
    Image TEXT,
    Title TEXT NOT NULL,
    Description TEXT,
    Item_Type TEXT,
    POS_Item INTGER,
    Reorder INTEGER,
    Deliver INTEGER,
    Pickup INTEGER,
    Buy INTEGER,
    Date DATE
);


CREATE TABLE Customer (
	Customer_ID INTEGER PRIMARY KEY,
	First_Name TEXT,
	Last_Name TEXT,
	Gender TEXT,
	Alias TEXT,
	Birth DATE,
	House_No TEXT,
	Street TEXT,
	Barangay_ID INT,
	City_ID INT,
	Province TEXT, 
	Item_ID INT,
	Client_Gallon INT,
	WRS_Gallon INT,
	Quantity INT,
	FOREIGN KEY (Barangay_ID) REFERENCES Barangay(Barangay_ID),
	FOREIGN KEY (City_ID) REFERENCES City(City_ID),
	FOREIGN KEY (Item_ID) REFERENCES Item(Item_ID)
)


CREATE TABLE AccountType (
    Account_Type_ID INTEGER PRIMARY KEY,
    Title TEXT NOT NULL,
    Description TEXT,
    Menu TEXT
);


CREATE TABLE UserAccount (
    User_Account_ID INTEGER PRIMARY KEY,
    Image TEXT,
    Firstname TEXT NOT NULL,
    Middlename TEXT,
    Lastname TEXT NOT NULL,
    Username TEXT,
    Email TEXT,
    Password TEXT,
    Contact TEXT,
    Gender TEXT,
    Address TEXT,
    Account_Type_ID INTEGER NOT NULL,
    Status TEXT CHECK(Status IN ('Active', 'Inactive')) DEFAULT 'Active',
    FOREIGN KEY (Account_Type_ID) REFERENCES AccountType(Account_Type_ID)
);
